```bash

const { spawn } = require('child_process');
const env = { 
  ...process.env, 
  OPENCODE_HOST: '192.168.255.255',

};
const child = spawn('node', ['dist/index.js'], { env, stdio: 'pipe', cwd: process.cwd() });
"
```
